## Context

P05-01 created the `ChipRegistry` and `ChipDefinition` types. P05-16 created the HDL compiler that produces evaluable `ChipDefinition` objects. This ticket creates the unified evaluation entry point that handles:
- Builtin chips → call evaluate function directly
- HDL-compiled chips → evaluate part graph recursively
- Circuit chips (future) → deserialize and evaluate

The nand2tetris pedagogy builds chips hierarchically:
```
Nand (builtin) → Not → And → Or → Xor → Mux → DMux → Not16 → ... → DMux8Way
```

Each chip uses previously-built chips as parts. The evaluator must recursively descend into chip internals.

---

## Goal

After this ticket:
1. `evaluateChip(chipDef, inputs)` → correct outputs for any chip type
2. Builtin: direct evaluate call
3. HDL: evaluate parts in topological order, recursing into sub-chips
4. Depth guard: max 100 levels, clean error on circular reference
5. Progressive hierarchy test: Nand → Not → And → Or → all correct

## Files to Create

### `src/core/chips/evaluate.ts`

```typescript
import type { ChipDefinition } from './types'
import { isBuiltinChip } from './types'

export class ChipEvaluationError extends Error {
  constructor(message: string, public chipName: string, public depth: number) {
    super(message)
    this.name = 'ChipEvaluationError'
  }
}

export const MAX_EVAL_DEPTH = 100

export function evaluateChip(
  chipDef: ChipDefinition,
  inputs: Record<string, number>,
  depth = 0
): Record<string, number> {
  if (depth > MAX_EVAL_DEPTH) {
    throw new ChipEvaluationError(
      `Maximum evaluation depth (${MAX_EVAL_DEPTH}) exceeded — possible circular reference`,
      chipDef.name,
      depth
    )
  }

  if (isBuiltinChip(chipDef)) {
    return chipDef.implementation.evaluate(inputs)
  }

  // **Contract with P05-16 (pick one and document in P05-16):**
  // A) Compiler emits `type: 'builtin'` only; generated `evaluate` calls `evaluateChip(part, ..., depth + 1)` for each part → this branch is unused.
  // B) Compiler emits `type: 'hdl'` with a part list / wires → implement topological eval here (this ticket owns full graph evaluation).
  if (chipDef.implementation.type === 'hdl') {
    throw new ChipEvaluationError(
      `HDL part-graph evaluation not implemented — align P05-16 to (A) or implement (B) in this file`,
      chipDef.name,
      depth
    )
  }

  if (chipDef.implementation.type === 'circuit') {
    // Future: deserialize circuit, build evaluation graph, evaluate
    // This comes with P05-26 (interop)
    throw new ChipEvaluationError(
      `Circuit chip evaluation not yet implemented`,
      chipDef.name,
      depth
    )
  }

  throw new ChipEvaluationError(
    `Unknown implementation type for chip "${chipDef.name}"`,
    chipDef.name,
    depth
  )
}
```

**Design note:** **Recommended:** P05-16 compiles user HDL to `type: 'builtin'` where the generated `evaluate` body calls `evaluateChip(childDef, mappedInputs, depth + 1)` for each sub-chip. Then this file stays a **dispatcher + depth guard** only. If P05-16 instead stores `type: 'hdl'`, P05-18 must implement part-graph evaluation (same depth threading).

### `src/core/chips/evaluate.test.ts`

```typescript
import { describe, it, expect, beforeEach } from 'vitest'
import { createChipRegistry, registerBuiltin } from './registry'
import type { ChipRegistry } from './registry'
import { evaluateChip, ChipEvaluationError, MAX_EVAL_DEPTH } from './evaluate'
// Paths depend on P05-04 / P05-16 layout — adjust imports to match merged code.
import { parseHDL } from '../hdl/parser'
import { compileHDL } from '../hdl/compiler'

let registry: ChipRegistry

beforeEach(() => {
  registry = createChipRegistry()
  // Nand builtin
  registerBuiltin(registry, 'Nand',
    [{ name: 'a', width: 1 }, { name: 'b', width: 1 }],
    [{ name: 'out', width: 1 }],
    (i) => ({ out: (i.a & i.b) === 0 ? 1 : 0 })
  )
})

function compileAndRegister(hdl: string): void {
  const parseResult = parseHDL(hdl)
  if (parseResult.type !== 'success') throw new Error('Parse failed')
  const compileResult = compileHDL(parseResult.chip, registry)
  if (compileResult.type !== 'success') throw new Error('Compile failed: ' + compileResult.errors.map(e => e.message).join(', '))
  registry.register(compileResult.chip)
}

describe('evaluateChip', () => {
  it('evaluates Nand builtin — all 4 combos', () => {
    const nand = registry.get('Nand')!
    expect(evaluateChip(nand, { a: 0, b: 0 })).toEqual({ out: 1 })
    expect(evaluateChip(nand, { a: 0, b: 1 })).toEqual({ out: 1 })
    expect(evaluateChip(nand, { a: 1, b: 0 })).toEqual({ out: 1 })
    expect(evaluateChip(nand, { a: 1, b: 1 })).toEqual({ out: 0 })
  })

  it('evaluates Not (compiled from HDL, uses Nand)', () => {
    compileAndRegister(`
      CHIP Not { IN in; OUT out;
        PARTS: Nand(a=in, b=in, out=out);
      }
    `)
    const not = registry.get('Not')!
    expect(evaluateChip(not, { in: 0 })).toEqual({ out: 1 })
    expect(evaluateChip(not, { in: 1 })).toEqual({ out: 0 })
  })

  it('evaluates And (compiled from HDL, uses Nand + Not)', () => {
    compileAndRegister('CHIP Not { IN in; OUT out; PARTS: Nand(a=in, b=in, out=out); }')
    compileAndRegister(`
      CHIP And { IN a, b; OUT out;
        PARTS:
        Nand(a=a, b=b, out=nandOut);
        Not(in=nandOut, out=out);
      }
    `)
    const and = registry.get('And')!
    expect(evaluateChip(and, { a: 0, b: 0 })).toEqual({ out: 0 })
    expect(evaluateChip(and, { a: 0, b: 1 })).toEqual({ out: 0 })
    expect(evaluateChip(and, { a: 1, b: 0 })).toEqual({ out: 0 })
    expect(evaluateChip(and, { a: 1, b: 1 })).toEqual({ out: 1 })
  })

  it('evaluates Or (3-level: uses Not + Nand)', () => {
    compileAndRegister('CHIP Not { IN in; OUT out; PARTS: Nand(a=in, b=in, out=out); }')
    compileAndRegister(`
      CHIP Or { IN a, b; OUT out;
        PARTS:
        Not(in=a, out=notA);
        Not(in=b, out=notB);
        Nand(a=notA, b=notB, out=out);
      }
    `)
    const or = registry.get('Or')!
    expect(evaluateChip(or, { a: 0, b: 0 })).toEqual({ out: 0 })
    expect(evaluateChip(or, { a: 0, b: 1 })).toEqual({ out: 1 })
    expect(evaluateChip(or, { a: 1, b: 0 })).toEqual({ out: 1 })
    expect(evaluateChip(or, { a: 1, b: 1 })).toEqual({ out: 1 })
  })

  it('depth guard triggers when starting depth already exceeds maximum', () => {
    const nand = registry.get('Nand')!
    expect(() => evaluateChip(nand, { a: 0, b: 0 }, MAX_EVAL_DEPTH + 1)).toThrow(ChipEvaluationError)
  })

  // Circular / mutual recursion: only reliable once P05-16 threads `depth + 1` into every nested
  // `evaluateChip` call; otherwise add an integration test with two compiled chips that call each other.
})
```

## Pitfalls

- **P05-16 design choice**: If the compiler stores compiled chips as `type: 'builtin'` (recommended), the hierarchy evaluation is already handled by the recursive closures. This file becomes a thin dispatcher + depth guard. If `type: 'hdl'`, implement full part-graph eval here.
- **Depth guard placement**: The depth counter must be threaded through recursive calls. If using closures, pass `depth + 1` when calling part chips. The closure generated by P05-16 should accept an optional `depth` parameter.
- **Compilation order matters**: `compileHDL` for And requires Not to already be in the registry. Tests must compile in dependency order: Not → And → Or.
- **No store dependency**: `evaluateChip` is a pure function — it does not touch Zustand. Input: `ChipDefinition` + `Record<string, number>`. Output: `Record<string, number>`. This makes it testable without store reset boilerplate.
