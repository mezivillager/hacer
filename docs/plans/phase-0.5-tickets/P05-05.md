## Context

Nand2tetris uses `.tst` test scripts to verify chip implementations. Each script loads a chip, sets input values, evaluates, records outputs, and compares against expected results.

HACER has no test infrastructure. The directory `src/core/testing/` does not exist.

**Important finding:** Project 1 `.tst` files do NOT use `repeat` or `while` — they are all explicit `set`/`eval`/`output` sequences. This simplifies the parser significantly. Support for `repeat`/`while` can be deferred to Phase 0.6.

## Goal

## TST Grammar (Project 1 scope)

```
Script     = Directive* Command*
Directive  = LoadDir | OutputFileDir | CompareDir | OutputListDir
LoadDir    = 'load' Filename ','
OutputFileDir = 'output-file' Filename ','
CompareDir = 'compare-to' Filename ','
OutputListDir = 'output-list' ColSpec+ ';'
ColSpec    = Name FormatSpec?
FormatSpec = '%' Format Padding '.' Width '.' Padding
Format     = 'B' | 'D' | 'X' | 'S'
Command    = SetCmd | EvalCmd | OutputCmd
SetCmd     = 'set' Name Value ','
EvalCmd    = 'eval' ','
OutputCmd  = 'output' ';'
Value      = Number | '%B' BinaryDigits | '%X' HexDigits | '%D' Number
```

**Key:** Commands end with `,` (comma), except `output` which ends with `;` (semicolon). The last command before `output;` typically has a `,` too.

## Parsing Strategy

The TST format is simpler than HDL — it's essentially a flat sequence of commands, not a nested structure. A line-oriented parser with command recognition works well:

1. **Strip comments** (`//` to end of line)
2. **Tokenize**: split on whitespace and punctuation (`,`, `;`, `%`)
3. **Identify commands** by keyword: `load`, `output-file`, `compare-to`, `output-list`, `set`, `eval`, `output`
4. **Parse `set` values**: handle decimal (`42`), binary (`%B0001001000110100`), hex (`%X1234`)
5. **Parse `output-list` format specifiers**: `a%B1.16.1` → name=`a`, format=`B`, padLeft=1, width=16, padRight=1

### Value parsing:

```typescript
function parseValue(token: string): number {
  if (token.startsWith('%B')) return parseInt(token.slice(2), 2)
  if (token.startsWith('%X')) return parseInt(token.slice(2), 16)
  if (token.startsWith('%D')) return parseInt(token.slice(2), 10)
  return parseInt(token, 10) // plain decimal
}
```

### Output-list format specifiers:

Format: `name%FP.W.P` where F=format, P=padding, W=width.
Simple format (no spec): `name` alone → default to `%B1.1.1` for 1-bit.

Examples from Project 1:
- `a b sel out` — simple names, no format specifiers (default 1-bit binary)
- `a%B1.16.1 b%B1.16.1 sel%B2.2.2 out%B1.16.1` — explicit 16-bit binary with padding

import { describe, it, expect } from 'vitest'
import { parseTST } from './tstParser'

const NAND_TST = `output-list a b out;

set a 0,
set b 0,
eval,
output;

set a 0,
set b 1,
eval,
output;

set a 1,
set b 0,
eval,
output;

set a 1,
set b 1,
eval,
output;`

const NOT_TST = `load Not.hdl,
compare-to Not.cmp,
output-list in out;

set in 0,
eval,
output;

set in 1,
eval,
output;`

const MUX4WAY16_TST = `load Mux4Way16.hdl,
compare-to Mux4Way16.cmp,
output-list a%B1.16.1 b%B1.16.1 c%B1.16.1 d%B1.16.1 sel%B2.2.2 out%B1.16.1;

set a 0,
set b 0,
set c 0,
set d 0,
set sel 0,
eval,
output;

set sel 1,
eval,
output;`

describe('TST Parser', () => {
  describe('basic parsing', () => {
    it('parses Nand.tst (no load, just output-list + commands)', () => {
      const result = parseTST(NAND_TST)
      expect(result.success).toBe(true)
      if (!result.success) return
      const cmds = result.script.commands

      // output-list with 3 columns
      const ol = cmds.find(c => c.type === 'output-list')
      expect(ol).toBeDefined()
      if (ol?.type === 'output-list') {
        expect(ol.columns).toHaveLength(3)
        expect(ol.columns.map(c => c.name)).toEqual(['a', 'b', 'out'])
      }

      // 4 groups of set+set+eval+output = 12 set commands, 4 evals, 4 outputs
      const sets = cmds.filter(c => c.type === 'set')
      expect(sets).toHaveLength(8) // 2 sets per test case × 4 cases
      const evals = cmds.filter(c => c.type === 'eval')
      expect(evals).toHaveLength(4)
      const outputs = cmds.filter(c => c.type === 'output')
      expect(outputs).toHaveLength(4)
    })

    it('parses Not.tst with load and compare-to directives', () => {
      const result = parseTST(NOT_TST)
      expect(result.success).toBe(true)
      if (!result.success) return
      const cmds = result.script.commands

      expect(cmds[0]).toEqual({ type: 'load', filename: 'Not.hdl' })
      expect(cmds[1]).toEqual({ type: 'compare-to', filename: 'Not.cmp' })
    })
  })

  describe('format specifiers', () => {
    it('parses output-list with explicit format specifiers', () => {
      const result = parseTST(MUX4WAY16_TST)
      expect(result.success).toBe(true)
      if (!result.success) return
      const ol = result.script.commands.find(c => c.type === 'output-list')
      if (ol?.type !== 'output-list') return

      expect(ol.columns[0]).toEqual({
        name: 'a', format: 'B', padLeft: 1, width: 16, padRight: 1,
      })
      expect(ol.columns[4]).toEqual({
        name: 'sel', format: 'B', padLeft: 2, width: 2, padRight: 2,
      })
    })

    it('defaults format for plain names (no % specifier)', () => {
      const result = parseTST(`output-list a b out;`)
      expect(result.success).toBe(true)
      if (!result.success) return
      const ol = result.script.commands.find(c => c.type === 'output-list')
      if (ol?.type !== 'output-list') return
      // Default: 1-bit binary
      expect(ol.columns[0].format).toBe('B')
    })
  })

  describe('value parsing', () => {
    it('parses decimal set values', () => {
      const result = parseTST(`output-list a;\nset a 42,\neval,\noutput;`)
      expect(result.success).toBe(true)
      if (!result.success) return
      const set = result.script.commands.find(c => c.type === 'set')
      if (set?.type === 'set') {
        expect(set.value).toBe(42)
      }
    })

    it('parses binary set values (%B prefix)', () => {
      const result = parseTST(`output-list a;\nset a %B0001001000110100,\neval,\noutput;`)
      expect(result.success).toBe(true)
      if (!result.success) return
      const set = result.script.commands.find(c => c.type === 'set')
      if (set?.type === 'set') {
        expect(set.value).toBe(0b0001001000110100)
      }
    })
  })

  describe('comments', () => {
    it('ignores line comments in test scripts', () => {
      const result = parseTST(`// Test Not
output-list in out;
set in 0,  // set to zero
eval,
output;`)
      expect(result.success).toBe(true)
    })
  })

  describe('all Project 1 TST files parse', () => {

      it(`parses ${name}.tst without errors`, () => {
        const result = parseTST(tst)
        expect(result.success).toBe(true)
      })
    }
  })

## Pitfalls

- **Command terminators**: Most commands end with `,` (comma). `output` ends with `;` (semicolon). The last `set` before `eval` has a `,`. Don't confuse terminators.
- **`output-list` format specifiers**: `a%B1.16.1` is ONE token — no space between name and `%`. Some output-lists have no specifiers (just space-separated names like `a b sel out`).
- **Nand.tst has no `load` directive** — it starts directly with `output-list`. Handle missing `load`.
- **Values can be negative** — not in Project 1, but good to note. For now, all values are non-negative integers.
- **`src/core/testing/` is a new directory tree** — create it entirely.
- **Do NOT implement `repeat`/`while`** — not needed for Project 1. If encountered, return a parse error.
