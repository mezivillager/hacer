## Context

P05-01 created the `ChipRegistry` + `registerBuiltin` helper and registered Nand as the first builtin. This ticket adds the remaining 15 Project 1 chips.

All builtins use `BuiltinEvalFn = (inputs: Record<string, number>) => Record<string, number>`. Values are integer bitmasks: single-bit pins use 0/1, 16-bit pins use 0–0xFFFF.

### Project 1 chips (from gap-analysis.md):

| Category | Chips |
|----------|-------|
| Single-bit | Not, And, Or, Xor, Mux, DMux |
| 16-bit | Not16, And16, Or16, Mux16 |
| Multi-way | Or8Way, Mux4Way16, Mux8Way16, DMux4Way, DMux8Way |

### Pin interfaces (reference):

| Chip | Inputs | Outputs |
|------|--------|---------|
| Not | `in` | `out` |
| And | `a, b` | `out` |
| Or | `a, b` | `out` |
| Xor | `a, b` | `out` |
| Mux | `a, b, sel` | `out` |
| DMux | `in, sel` | `a, b` |
| Not16 | `in[16]` | `out[16]` |
| And16 | `a[16], b[16]` | `out[16]` |
| Or16 | `a[16], b[16]` | `out[16]` |
| Mux16 | `a[16], b[16], sel` | `out[16]` |
| Or8Way | `in[8]` | `out` |
| Mux4Way16 | `a[16], b[16], c[16], d[16], sel[2]` | `out[16]` |
| Mux8Way16 | `a[16], b[16], c[16], d[16], e[16], f[16], g[16], h[16], sel[3]` | `out[16]` |
| DMux4Way | `in, sel[2]` | `a, b, c, d` |
| DMux8Way | `in, sel[3]` | `a, b, c, d, e, f, g, h` |

## Files to Create

### `src/core/chips/builtins/project01.ts`

```typescript
import type { ChipRegistry } from '../registry'
import { registerBuiltin } from '../registry'
import type { ChipPin } from '../types'

const MASK16 = 0xFFFF

function pin(name: string, width = 1): ChipPin {
  return { name, width }
}

export function registerProject1Builtins(registry: ChipRegistry): void {
  // Nand already registered by P05-01 — skip if present
  if (!registry.has('Nand')) {
    registerBuiltin(registry, 'Nand',
      [pin('a'), pin('b')], [pin('out')],
      (i) => ({ out: (i.a & i.b) === 0 ? 1 : 0 })
    )
  }

  registerBuiltin(registry, 'Not',
    [pin('in')], [pin('out')],
    (i) => ({ out: i.in === 0 ? 1 : 0 })
  )

  registerBuiltin(registry, 'And',
    [pin('a'), pin('b')], [pin('out')],
    (i) => ({ out: i.a & i.b })
  )

  registerBuiltin(registry, 'Or',
    [pin('a'), pin('b')], [pin('out')],
    (i) => ({ out: (i.a | i.b) & 1 })
  )

  registerBuiltin(registry, 'Xor',
    [pin('a'), pin('b')], [pin('out')],
    (i) => ({ out: (i.a ^ i.b) & 1 })
  )

  registerBuiltin(registry, 'Mux',
    [pin('a'), pin('b'), pin('sel')], [pin('out')],
    (i) => ({ out: i.sel === 0 ? i.a : i.b })
  )

  registerBuiltin(registry, 'DMux',
    [pin('in'), pin('sel')], [pin('a'), pin('b')],
    (i) => ({ a: i.sel === 0 ? i.in : 0, b: i.sel === 1 ? i.in : 0 })
  )

  // 16-bit chips
  registerBuiltin(registry, 'Not16',
    [pin('in', 16)], [pin('out', 16)],
    (i) => ({ out: (~i.in & MASK16) >>> 0 })
  )

  registerBuiltin(registry, 'And16',
    [pin('a', 16), pin('b', 16)], [pin('out', 16)],
    (i) => ({ out: (i.a & i.b & MASK16) >>> 0 })
  )

  registerBuiltin(registry, 'Or16',
    [pin('a', 16), pin('b', 16)], [pin('out', 16)],
    (i) => ({ out: ((i.a | i.b) & MASK16) >>> 0 })
  )

  registerBuiltin(registry, 'Mux16',
    [pin('a', 16), pin('b', 16), pin('sel')], [pin('out', 16)],
    (i) => ({ out: (i.sel === 0 ? i.a : i.b) & MASK16 })
  )

  // Multi-way chips
  registerBuiltin(registry, 'Or8Way',
    [pin('in', 8)], [pin('out')],
    (i) => ({ out: i.in !== 0 ? 1 : 0 })
  )

  registerBuiltin(registry, 'Mux4Way16',
    [pin('a', 16), pin('b', 16), pin('c', 16), pin('d', 16), pin('sel', 2)],
    [pin('out', 16)],
    (i) => ({ out: [i.a, i.b, i.c, i.d][i.sel & 0x3] & MASK16 })
  )

  registerBuiltin(registry, 'Mux8Way16',
    [pin('a', 16), pin('b', 16), pin('c', 16), pin('d', 16),
     pin('e', 16), pin('f', 16), pin('g', 16), pin('h', 16), pin('sel', 3)],
    [pin('out', 16)],
    (i) => ({ out: [i.a, i.b, i.c, i.d, i.e, i.f, i.g, i.h][i.sel & 0x7] & MASK16 })
  )

  registerBuiltin(registry, 'DMux4Way',
    [pin('in'), pin('sel', 2)],
    [pin('a'), pin('b'), pin('c'), pin('d')],
    (i) => {
      const result = { a: 0, b: 0, c: 0, d: 0 }
      const keys = ['a', 'b', 'c', 'd'] as const
      result[keys[i.sel & 0x3]] = i.in
      return result
    }
  )

  registerBuiltin(registry, 'DMux8Way',
    [pin('in'), pin('sel', 3)],
    [pin('a'), pin('b'), pin('c'), pin('d'), pin('e'), pin('f'), pin('g'), pin('h')],
    (i) => {
      const result = { a: 0, b: 0, c: 0, d: 0, e: 0, f: 0, g: 0, h: 0 }
      const keys = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h'] as const
      result[keys[i.sel & 0x7]] = i.in
      return result
    }
  )
}
```

### `src/core/chips/builtins/project01.test.ts`

```typescript
import { describe, it, expect, beforeEach } from 'vitest'
import { createChipRegistry } from '../registry'
import { registerProject1Builtins } from './project01'
import { isBuiltinChip } from '../types'
import type { ChipRegistry } from '../registry'

let registry: ChipRegistry

beforeEach(() => {
  registry = createChipRegistry()
  registerProject1Builtins(registry)
})

describe('registerProject1Builtins', () => {
  it('registers all 16 chips', () => {
    expect(registry.list()).toHaveLength(16)
  })

  const chipNames = [
    'Nand', 'Not', 'And', 'Or', 'Xor', 'Mux', 'DMux',
    'Not16', 'And16', 'Or16', 'Mux16',
    'Or8Way', 'Mux4Way16', 'Mux8Way16', 'DMux4Way', 'DMux8Way',
  ]

  for (const name of chipNames) {
    it(`registers ${name}`, () => {
      expect(registry.has(name)).toBe(true)
    })
  }
})

describe('single-bit builtins', () => {
  function evalChip(name: string, inputs: Record<string, number>): Record<string, number> {
    const chip = registry.get(name)!
    if (!isBuiltinChip(chip)) throw new Error(`${name} is not builtin`)
    return chip.implementation.evaluate(inputs)
  }

  it('Not: 0→1, 1→0', () => {
    expect(evalChip('Not', { in: 0 })).toEqual({ out: 1 })
    expect(evalChip('Not', { in: 1 })).toEqual({ out: 0 })
  })

  it('And: all 4 combos', () => {
    expect(evalChip('And', { a: 0, b: 0 })).toEqual({ out: 0 })
    expect(evalChip('And', { a: 0, b: 1 })).toEqual({ out: 0 })
    expect(evalChip('And', { a: 1, b: 0 })).toEqual({ out: 0 })
    expect(evalChip('And', { a: 1, b: 1 })).toEqual({ out: 1 })
  })

  it('Or: all 4 combos', () => {
    expect(evalChip('Or', { a: 0, b: 0 })).toEqual({ out: 0 })
    expect(evalChip('Or', { a: 0, b: 1 })).toEqual({ out: 1 })
    expect(evalChip('Or', { a: 1, b: 0 })).toEqual({ out: 1 })
    expect(evalChip('Or', { a: 1, b: 1 })).toEqual({ out: 1 })
  })

  it('Xor: all 4 combos', () => {
    expect(evalChip('Xor', { a: 0, b: 0 })).toEqual({ out: 0 })
    expect(evalChip('Xor', { a: 0, b: 1 })).toEqual({ out: 1 })
    expect(evalChip('Xor', { a: 1, b: 0 })).toEqual({ out: 1 })
    expect(evalChip('Xor', { a: 1, b: 1 })).toEqual({ out: 0 })
  })

  it('Mux: sel=0→a, sel=1→b', () => {
    expect(evalChip('Mux', { a: 1, b: 0, sel: 0 })).toEqual({ out: 1 })
    expect(evalChip('Mux', { a: 1, b: 0, sel: 1 })).toEqual({ out: 0 })
  })

  it('DMux: sel=0→a=in, sel=1→b=in', () => {
    expect(evalChip('DMux', { in: 1, sel: 0 })).toEqual({ a: 1, b: 0 })
    expect(evalChip('DMux', { in: 1, sel: 1 })).toEqual({ a: 0, b: 1 })
  })
})

describe('16-bit builtins', () => {
  function evalChip(name: string, inputs: Record<string, number>): Record<string, number> {
    const chip = registry.get(name)!
    if (!isBuiltinChip(chip)) throw new Error(`${name} is not builtin`)
    return chip.implementation.evaluate(inputs)
  }

  it('Not16: inverts all bits', () => {
    expect(evalChip('Not16', { in: 0x0000 })).toEqual({ out: 0xFFFF })
    expect(evalChip('Not16', { in: 0xFFFF })).toEqual({ out: 0x0000 })
    expect(evalChip('Not16', { in: 0x1234 })).toEqual({ out: 0xEDCB })
  })

  it('And16: bitwise AND', () => {
    expect(evalChip('And16', { a: 0xFF00, b: 0x0FF0 })).toEqual({ out: 0x0F00 })
  })

  it('Or16: bitwise OR', () => {
    expect(evalChip('Or16', { a: 0xFF00, b: 0x00FF })).toEqual({ out: 0xFFFF })
  })

  it('Mux16: sel=0→a, sel=1→b', () => {
    expect(evalChip('Mux16', { a: 0x1234, b: 0x5678, sel: 0 })).toEqual({ out: 0x1234 })
    expect(evalChip('Mux16', { a: 0x1234, b: 0x5678, sel: 1 })).toEqual({ out: 0x5678 })
  })
})

describe('multi-way builtins', () => {
  function evalChip(name: string, inputs: Record<string, number>): Record<string, number> {
    const chip = registry.get(name)!
    if (!isBuiltinChip(chip)) throw new Error(`${name} is not builtin`)
    return chip.implementation.evaluate(inputs)
  }

  it('Or8Way: any bit set → 1', () => {
    expect(evalChip('Or8Way', { in: 0 })).toEqual({ out: 0 })
    expect(evalChip('Or8Way', { in: 0x01 })).toEqual({ out: 1 })
    expect(evalChip('Or8Way', { in: 0xFF })).toEqual({ out: 1 })
  })

  it('Mux4Way16: selects correct input', () => {
    const inputs = { a: 0x1111, b: 0x2222, c: 0x3333, d: 0x4444 }
    expect(evalChip('Mux4Way16', { ...inputs, sel: 0 })).toEqual({ out: 0x1111 })
    expect(evalChip('Mux4Way16', { ...inputs, sel: 1 })).toEqual({ out: 0x2222 })
    expect(evalChip('Mux4Way16', { ...inputs, sel: 2 })).toEqual({ out: 0x3333 })
    expect(evalChip('Mux4Way16', { ...inputs, sel: 3 })).toEqual({ out: 0x4444 })
  })

  it('DMux4Way: routes input to selected output', () => {
    expect(evalChip('DMux4Way', { in: 1, sel: 0 })).toEqual({ a: 1, b: 0, c: 0, d: 0 })
    expect(evalChip('DMux4Way', { in: 1, sel: 2 })).toEqual({ a: 0, b: 0, c: 1, d: 0 })
  })

  it('DMux8Way: routes input to 1 of 8 outputs', () => {
    const result = evalChip('DMux8Way', { in: 1, sel: 5 })
    expect(result.f).toBe(1)
    expect(result.a + result.b + result.c + result.d + result.e + result.g + result.h).toBe(0)
  })
})
```

## Pitfalls

- **Bitmask arithmetic**: Use `>>> 0` for unsigned right shift and `& MASK16` to clamp to 16 bits. JavaScript bitwise ops produce signed 32-bit integers — `~0xFFFF` is `-65536`, not `0xFFFF0000`.
- **Nand already registered**: If the registry already has Nand from P05-01 tests, `registerProject1Builtins` must guard against duplicate registration (check `registry.has('Nand')` first).
- **DMux truth table**: DMux4Way with `sel=2` routes to output `c` (0-indexed: a=0, b=1, c=2, d=3). DMux8Way follows the same pattern with 3-bit sel.
- **Or8Way input width**: `in[8]` means the input is 8 bits. `Or8Way` returns 1 if any bit is set. Use `in !== 0 ? 1 : 0`.
- **Don't modify gateLogic.ts**: The registry builtins are separate from the existing `gateLogic` lookup table. They coexist.
