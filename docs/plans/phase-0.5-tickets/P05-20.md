## Context

P05-01 created `ChipDefinition` with named typed pins. P05-08 added rename actions and name display on 3D nodes. But there's no UI panel to:
- Declare "I am building chip X"
- See the required pin interface
- Auto-create I/O nodes matching the spec
- Validate that the circuit's I/O matches before running tests

### Current workflow gap:

A student selects a chip from the workflow browser (P05-19), but there's no formal binding between the chip spec and the actual I/O nodes on the canvas. They might have extra nodes, missing nodes, or misnamed nodes — and the test engine would fail with cryptic errors.

### Chip definition needs to track:

```typescript
// In store state:
currentChipDefinition: ChipDefinition | null
```

When set, it defines the "target" chip the student is building. The panel displays this target and validates the canvas against it.

### `src/core/chips/validation.ts`

```typescript
import type { ChipDefinition } from './types'
import type { CircuitState } from '@/store/types'

export interface IOValidationError {
  type: 'missing_input' | 'missing_output' | 'extra_input' | 'extra_output' | 'width_mismatch'
  pinName: string
  expected?: number // expected width
  actual?: number   // actual width
  message: string
}

export function validateCircuitIO(
  state: CircuitState,
  chipDef: ChipDefinition
): IOValidationError[] {
  const errors: IOValidationError[] = []

  // Check required inputs exist with correct widths
  for (const requiredInput of chipDef.inputs) {
    const found = state.inputNodes.find(n => n.name === requiredInput.name)
    if (!found) {
      errors.push({
        type: 'missing_input',
        pinName: requiredInput.name,
        expected: requiredInput.width,
        message: `Missing input node: "${requiredInput.name}"`,
      })
    } else if (found.width !== requiredInput.width) {
      errors.push({
        type: 'width_mismatch',
        pinName: requiredInput.name,
        expected: requiredInput.width,
        actual: found.width,
        message: `Input "${requiredInput.name}" has width ${found.width}, expected ${requiredInput.width}`,
      })
    }
  }

  // Check required outputs exist with correct widths
  for (const requiredOutput of chipDef.outputs) {
    const found = state.outputNodes.find(n => n.name === requiredOutput.name)
    if (!found) {
      errors.push({
        type: 'missing_output',
        pinName: requiredOutput.name,
        expected: requiredOutput.width,
        message: `Missing output node: "${requiredOutput.name}"`,
      })
    } else if (found.width !== requiredOutput.width) {
      errors.push({
        type: 'width_mismatch',
        pinName: requiredOutput.name,
        expected: requiredOutput.width,
        actual: found.width,
        message: `Output "${requiredOutput.name}" has width ${found.width}, expected ${requiredOutput.width}`,
      })
    }
  }

  // Check for extra nodes not in the spec
  const expectedInputNames = new Set(chipDef.inputs.map(p => p.name))
  for (const node of state.inputNodes) {
    if (!expectedInputNames.has(node.name)) {
      errors.push({
        type: 'extra_input',
        pinName: node.name,
        message: `Extra input node: "${node.name}" (not in chip spec)`,
      })
    }
  }

  const expectedOutputNames = new Set(chipDef.outputs.map(p => p.name))
  for (const node of state.outputNodes) {
    if (!expectedOutputNames.has(node.name)) {
      errors.push({
        type: 'extra_output',
        pinName: node.name,
        message: `Extra output node: "${node.name}" (not in chip spec)`,
      })
    }
  }

  return errors
}
```

### `src/core/chips/validation.test.ts`

```typescript
import { describe, it, expect } from 'vitest'
import { validateCircuitIO } from './validation'
import type { ChipDefinition } from './types'
import type { CircuitState } from '@/store/types'

function makeChipDef(
  inputs: Array<{ name: string; width: number }>,
  outputs: Array<{ name: string; width: number }>
): ChipDefinition {
  return {
    name: 'TestChip',
    inputs,
    outputs,
    implementation: { type: 'builtin', evaluate: () => ({}) },
  }
}

function makeState(
  inputNodes: Array<{ name: string; width: number }>,
  outputNodes: Array<{ name: string; width: number }>
): CircuitState {
  return {
    gates: [], wires: [], selectedGateId: null, selectedWireId: null,
    simulationRunning: false, simulationSpeed: 100, placementMode: null,
    placementPreviewPosition: null, wiringFrom: null, isDragActive: false,
    hoveredGateId: null, showAxes: false, junctions: [],
    nodePlacementMode: null, selectedNodeId: null, selectedNodeType: null,
    junctionPlacementMode: null, junctionPreviewPosition: null, junctionPreviewWireId: null,
    inputNodes: inputNodes.map((n, i) => ({
      id: `in-${i}`, name: n.name, position: { x: 0, y: 0, z: 0 },
      rotation: { x: 0, y: 0, z: 0 },       value: 0, width: n.width,
    })),
    outputNodes: outputNodes.map((n, i) => ({
      id: `out-${i}`, name: n.name, position: { x: 0, y: 0, z: 0 },
      rotation: { x: 0, y: 0, z: 0 }, value: 0, width: n.width,
    })),
  }
}

describe('validateCircuitIO', () => {
  it('valid: matching pins', () => {
    const chip = makeChipDef([{ name: 'a', width: 1 }], [{ name: 'out', width: 1 }])
    const state = makeState([{ name: 'a', width: 1 }], [{ name: 'out', width: 1 }])
    expect(validateCircuitIO(state, chip)).toEqual([])
  })

  it('detects missing input', () => {
    const chip = makeChipDef([{ name: 'a', width: 1 }, { name: 'b', width: 1 }], [{ name: 'out', width: 1 }])
    const state = makeState([{ name: 'a', width: 1 }], [{ name: 'out', width: 1 }])
    const errors = validateCircuitIO(state, chip)
    expect(errors).toHaveLength(1)
    expect(errors[0].type).toBe('missing_input')
    expect(errors[0].pinName).toBe('b')
  })

  it('detects width mismatch', () => {
    const chip = makeChipDef([{ name: 'in', width: 16 }], [{ name: 'out', width: 16 }])
    const state = makeState([{ name: 'in', width: 1 }], [{ name: 'out', width: 16 }])
    const errors = validateCircuitIO(state, chip)
    expect(errors).toHaveLength(1)
    expect(errors[0].type).toBe('width_mismatch')
  })

  it('detects extra node', () => {
    const chip = makeChipDef([{ name: 'a', width: 1 }], [{ name: 'out', width: 1 }])
    const state = makeState([{ name: 'a', width: 1 }, { name: 'extra', width: 1 }], [{ name: 'out', width: 1 }])
    const errors = validateCircuitIO(state, chip)
    expect(errors).toHaveLength(1)
    expect(errors[0].type).toBe('extra_input')
  })
})
```

### `src/store/actions/chipActions/chipActions.ts`

```typescript
import type { ChipDefinition } from '@/core/chips/types'
import type { Position } from '@/store/types'

export interface ChipActions {
  setCurrentChip: (chipDef: ChipDefinition | null) => void
  autoGenerateIO: () => void
}

export const createChipActions = (set: any, get: any): ChipActions => ({
  setCurrentChip: (chipDef: ChipDefinition | null) => {
    set((state: any) => {
      state.currentChipDefinition = chipDef
    }, false, 'setCurrentChip')
  },

  autoGenerateIO: () => {
    set((state: any) => {
      const chipDef = state.currentChipDefinition
      if (!chipDef) return

      // Clear existing I/O nodes
      state.inputNodes = []
      state.outputNodes = []

      // Create input nodes on the left
      chipDef.inputs.forEach((pin: any, i: number) => {
        state.inputNodes.push({
          id: `input-${pin.name}-${Date.now()}-${i}`,
          name: pin.name,
          position: { x: -4, y: 0, z: i * 4 },
          rotation: { x: 0, y: 0, z: 0 },
          value: 0,
          width: pin.width,
        })
      })

      // Create output nodes on the right
      chipDef.outputs.forEach((pin: any, i: number) => {
        state.outputNodes.push({
          id: `output-${pin.name}-${Date.now()}-${i}`,
          name: pin.name,
          position: { x: 8, y: 0, z: i * 4 },
          rotation: { x: 0, y: 0, z: 0 },
          value: 0,
          width: pin.width,
        })
      })
    }, false, 'autoGenerateIO')
  },
})
```

## Pitfalls

- **`currentChipDefinition` serialization**: This field holds a `ChipDefinition` which may contain closures (evaluate function). Do NOT serialize it to localStorage. It's transient UI state — rebuilt when a chip is selected.
- **`autoGenerateIO` / `clearCircuit` and wires**: Clearing or replacing I/O nodes without removing incident **wires** leaves dangling references. Prefer **`clearCircuit()`** (if it clears wires) or explicitly remove wires connected to old nodes before replacing I/O — mirror whatever P05-14 / wire handlers guarantee.
- **Validation vs UI severity**: `validateCircuitIO` returns a flat list; treat **`missing_*` and `width_mismatch` as blocking** for “Run test”, and **`extra_*` as warnings** in the panel (or show all with labels). Ticket text previously contradicted itself — the pure function does not encode severity; the UI does.
- **Integration with P05-19**: When `ChipWorkflowBrowser` calls `handleStart`, it should also call `setCurrentChip` with the chip's definition. Either merge the logic or have ChipWorkflowBrowser dispatch both actions.
- **`CircuitStore` type expansion**: Adding `ChipActions` and `currentChipDefinition` to the union requires updating both the interface and the store creation. Follow the existing pattern (GateActions, WireActions, etc.).
