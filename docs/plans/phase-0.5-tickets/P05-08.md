## Context

HACER's I/O nodes (`InputNode`, `OutputNode`) have a `name` field, but:

1. **No rename action** — names are auto-generated as `in0`, `in1`, `out0`, `out1` by `nodePlacementActions.ts` and cannot be changed. There is no `renameInputNode` or `renameOutputNode` action.

2. **Names not displayed** — Both `InputNode3D.tsx` and `OutputNode3D.tsx` accept a `name` prop but render only the current value (`"0"` or `"1"`) on the top face. The `name` prop is destructured but unused (`_name` pattern — the component receives it but doesn't render it).

Students need to see which node is `"a"` vs `"b"` vs `"sel"`, and must be able to rename nodes to match the chip specification for test scripts to work.

### Current code locations:
- **Node types**: `src/store/types.ts` lines 23–43 (`InputNode`, `OutputNode` with `name: string`)
- **Node actions**: `src/store/actions/nodeActions/nodeActions.ts` — has `addInputNode`, `addOutputNode`, `updateInputNodeValue`, `updateInputNodePosition`, `updateOutputNodePosition` but NO rename
- **Node actions type**: `src/store/types.ts` `NodeActions` interface — does not include rename
- **InputNode3D**: `src/nodes/components/InputNode3D.tsx` — renders `value ? '1' : '0'` as text on the node body
- **OutputNode3D**: `src/nodes/components/OutputNode3D.tsx` — same pattern

### `src/store/actions/nodeActions/nodeActions.ts`

Add two new actions:

```typescript
renameInputNode: (nodeId: string, newName: string) => {
  set((state) => {
    const node = state.inputNodes.find(n => n.id === nodeId)
    if (!node) return
    const trimmed = newName.trim()
    if (trimmed === '') return
    const duplicate = state.inputNodes.some(n => n.id !== nodeId && n.name === trimmed)
    if (duplicate) return
    node.name = trimmed
  }, false, 'renameInputNode')
},

renameOutputNode: (nodeId: string, newName: string) => {
  set((state) => {
    const node = state.outputNodes.find(n => n.id === nodeId)
    if (!node) return
    const trimmed = newName.trim()
    if (trimmed === '') return
    const duplicate = state.outputNodes.some(n => n.id !== nodeId && n.name === trimmed)
    if (duplicate) return
    node.name = trimmed
  }, false, 'renameOutputNode')
},
```

### `src/store/actions/nodeActions/renameNode.test.ts`

```typescript
import { describe, it, expect, beforeEach } from 'vitest'
import { useCircuitStore } from '@/store/circuitStore'

function getState() {
  return useCircuitStore.getState()
}

beforeEach(() => {
  useCircuitStore.setState({
    gates: [],
    wires: [],
    selectedGateId: null,
    selectedWireId: null,
    simulationRunning: false,
    simulationSpeed: 100,
    placementMode: null,
    placementPreviewPosition: null,
    wiringFrom: null,
    isDragActive: false,
    hoveredGateId: null,
    showAxes: false,
    inputNodes: [],
    outputNodes: [],
    junctions: [],
    nodePlacementMode: null,
    selectedNodeId: null,
    selectedNodeType: null,
    junctionPlacementMode: null,
    junctionPreviewPosition: null,
    junctionPreviewWireId: null,
  })
})

describe('renameInputNode', () => {
  it('renames an input node', () => {
    const node = getState().addInputNode('in0', { x: 0, y: 0, z: 0 })
    getState().renameInputNode(node.id, 'a')
    const updated = getState().inputNodes.find(n => n.id === node.id)
    expect(updated?.name).toBe('a')
  })

  it('trims whitespace from new name', () => {
    const node = getState().addInputNode('in0', { x: 0, y: 0, z: 0 })
    getState().renameInputNode(node.id, '  sel  ')
    expect(getState().inputNodes.find(n => n.id === node.id)?.name).toBe('sel')
  })

  it('rejects empty name', () => {
    const node = getState().addInputNode('in0', { x: 0, y: 0, z: 0 })
    getState().renameInputNode(node.id, '')
    expect(getState().inputNodes.find(n => n.id === node.id)?.name).toBe('in0')
  })

  it('rejects duplicate name within input nodes', () => {
    const a = getState().addInputNode('a', { x: 0, y: 0, z: 0 })
    getState().addInputNode('b', { x: 0, y: 1, z: 0 })
    getState().renameInputNode(a.id, 'b') // 'b' already exists
    expect(getState().inputNodes.find(n => n.id === a.id)?.name).toBe('a')
  })

  it('no-op for non-existent node', () => {
    getState().renameInputNode('nonexistent', 'x')
    // No error thrown, state unchanged
    expect(getState().inputNodes).toHaveLength(0)
  })
})

describe('renameOutputNode', () => {
  it('renames an output node', () => {
    const node = getState().addOutputNode('out0', { x: 0, y: 0, z: 0 })
    getState().renameOutputNode(node.id, 'out')
    expect(getState().outputNodes.find(n => n.id === node.id)?.name).toBe('out')
  })

  it('rejects duplicate name within output nodes', () => {
    const a = getState().addOutputNode('a', { x: 0, y: 0, z: 0 })
    getState().addOutputNode('b', { x: 0, y: 1, z: 0 })
    getState().renameOutputNode(a.id, 'b')
    expect(getState().outputNodes.find(n => n.id === a.id)?.name).toBe('a')
  })
})
```

## Pitfalls

- **Drei `<Text>` component**: Import from `@react-three/drei` (already a dependency). It renders bitmap text in the 3D scene. Use it for the name label — the existing components already use it for value display.
- **Position the name label above the node**: The node body is `NODE_DIMENSIONS.BODY_HEIGHT / 2` tall. Place the name text at `BODY_HEIGHT / 2 + 0.15` (above the top face).
- **Don't break existing rendering**: The value text stays. Add the name text as a sibling element. If space is tight, reduce font sizes slightly.
- **Duplicate check scope**: Input names are unique among inputs. Output names are unique among outputs. But an input and an output CAN have the same name (e.g., both `"a"` — though this is unusual).
- **The `circuitActions` object in `circuitStore.ts`**: New actions need to be added both to the store creation (in `createNodeActions`) and to the `circuitActions` facade. Look at how existing actions like `addInputNode` are wired.
- **E2E window globals**: If `renameInputNode` should be accessible via `window.__CIRCUIT_ACTIONS__`, add it to the circuitActions object (it's auto-exposed).
