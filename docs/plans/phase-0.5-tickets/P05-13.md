## Context

After P05-02, node values are `number`. Single-bit nodes toggle 0/1. But 16-bit nodes need a richer input method — nand2tetris test scripts use `%B` (binary), `%D` (decimal), and `%X` (hex) format specifiers.

The current `InputNode3D.tsx` shows `value ? '1' : '0'` (or `String(Number(value))` after P05-02) on the 3D mesh top face. There is no multi-bit editing workflow.

### Current node interaction:

`InputNode3D.tsx` toggles value on shift+click via `updateInputNodeValue`. `OutputNode3D.tsx` is read-only.

---

## Goal

After this ticket:
1. `MultiBitInput` component renders bit toggles for width ≤ 8, or a numeric text input for wider buses
2. `MultiBitDisplay` shows values in binary, decimal, or hex based on a format selector
3. Format selector (B/D/X) works for both input and output nodes
4. 3D node faces show formatted values: `0xFF` (hex), `255` (decimal), `0000000011111111` (binary)
5. Integrated into `PinoutPanel` (P05-10) for sidebar editing, and available for 3D node text labels

## Files to Create

### `src/components/ui/MultiBitInput.tsx`

```tsx
import { useCircuitStore } from '@/store/circuitStore'
import { Button } from 'antd'
import { useState } from 'react'

type DisplayFormat = 'B' | 'D' | 'X'

interface MultiBitInputProps {
  nodeId: string
  currentValue: number
  width: number
  onValueChange: (nodeId: string, value: number) => void
}

export function formatValue(value: number, width: number, format: DisplayFormat): string {
  const mask = width >= 32 ? 0xFFFFFFFF : (1 << width) - 1
  const clamped = (value & mask) >>> 0
  switch (format) {
    case 'B': return clamped.toString(2).padStart(width, '0')
    case 'D': return String(clamped)
    case 'X': return '0x' + clamped.toString(16).toUpperCase().padStart(Math.ceil(width / 4), '0')
  }
}

export function parseValue(text: string, width: number): number | null {
  let num: number
  const trimmed = text.trim()
  if (trimmed.startsWith('0x') || trimmed.startsWith('0X')) {
    num = parseInt(trimmed, 16)
  } else if (trimmed.startsWith('0b') || trimmed.startsWith('0B')) {
    num = parseInt(trimmed.slice(2), 2)
  } else {
    num = parseInt(trimmed, 10)
  }
  if (isNaN(num) || num < 0) return null
  const mask = width >= 32 ? 0xFFFFFFFF : (1 << width) - 1
  return (num & mask) >>> 0
}

export function MultiBitInput({ nodeId, currentValue, width, onValueChange }: MultiBitInputProps) {
  const [format, setFormat] = useState<DisplayFormat>('D')
  const [editText, setEditText] = useState('')
  const [isEditing, setIsEditing] = useState(false)

  if (width <= 8) {
    // Render individual bit toggles
    return (
      <div data-testid={`multibit-input-${nodeId}`}>
        <div style={{ display: 'flex', gap: 2 }}>
          {Array.from({ length: width }, (_, i) => {
            const bit = (currentValue >>> (width - 1 - i)) & 1
            return (
              <span
                key={i}
                data-testid={`bit-toggle-${i}`}
                style={{
                  cursor: 'pointer',
                  fontFamily: 'monospace',
                  fontSize: 11,
                  padding: '1px 3px',
                  background: bit ? '#1890ff' : '#333',
                  color: '#fff',
                  borderRadius: 2,
                }}
                onClick={() => {
                  const toggled = currentValue ^ (1 << (width - 1 - i))
                  onValueChange(nodeId, toggled)
                }}
              >
                {bit}
              </span>
            )
          })}
        </div>
        <FormatSelector format={format} onChange={setFormat} />
        <span style={{ fontFamily: 'monospace', fontSize: 11 }}>
          {formatValue(currentValue, width, format)}
        </span>
      </div>
    )
  }

  // Wider buses: numeric text input
  return (
    <div data-testid={`multibit-input-${nodeId}`}>
      {isEditing ? (
        <input
          data-testid="multibit-text-input"
          style={{ width: 80, fontFamily: 'monospace', fontSize: 11 }}
          value={editText}
          onChange={(e) => setEditText(e.target.value)}
          onBlur={() => {
            const parsed = parseValue(editText, width)
            if (parsed !== null) onValueChange(nodeId, parsed)
            setIsEditing(false)
          }}
          onKeyDown={(e) => {
            if (e.key === 'Enter') (e.target as HTMLInputElement).blur()
          }}
          autoFocus
        />
      ) : (
        <span
          style={{ cursor: 'pointer', fontFamily: 'monospace', fontSize: 11 }}
          onClick={() => { setEditText(formatValue(currentValue, width, format)); setIsEditing(true) }}
          data-testid="multibit-display"
        >
          {formatValue(currentValue, width, format)}
        </span>
      )}
      <FormatSelector format={format} onChange={setFormat} />
    </div>
  )
}

function FormatSelector({ format, onChange }: { format: DisplayFormat; onChange: (f: DisplayFormat) => void }) {
  return (
    <span data-testid="format-selector" style={{ marginLeft: 4 }}>
      {(['B', 'D', 'X'] as const).map(f => (
        <span
          key={f}
          data-testid={`format-${f}`}
          onClick={() => onChange(f)}
          style={{
            cursor: 'pointer',
            padding: '0 3px',
            fontWeight: f === format ? 700 : 400,
            color: f === format ? '#1890ff' : '#888',
            fontSize: 10,
          }}
        >
          {f}
        </span>
      ))}
    </span>
  )
}
```

### `src/components/ui/MultiBitInput.test.tsx`

```typescript
import { describe, it, expect, vi } from 'vitest'
import { render, screen, fireEvent } from '@testing-library/react'
import { MultiBitInput, formatValue, parseValue } from './MultiBitInput'

describe('formatValue', () => {
  it('binary format for 4-bit value', () => {
    expect(formatValue(0b1010, 4, 'B')).toBe('1010')
  })
  it('decimal format', () => {
    expect(formatValue(255, 8, 'D')).toBe('255')
  })
  it('hex format with padding', () => {
    expect(formatValue(0xFF, 16, 'X')).toBe('0x00FF')
  })
})

describe('parseValue', () => {
  it('parses decimal', () => expect(parseValue('42', 8)).toBe(42))
  it('parses hex', () => expect(parseValue('0xFF', 8)).toBe(255))
  it('parses binary', () => expect(parseValue('0b1010', 4)).toBe(10))
  it('clamps to width', () => expect(parseValue('256', 8)).toBe(0))
  it('returns null for invalid', () => expect(parseValue('abc', 8)).toBeNull())
})

describe('MultiBitInput (width ≤ 8)', () => {
  it('renders bit toggles for width=4', () => {
    const onChange = vi.fn()
    render(<MultiBitInput nodeId="n1" currentValue={0b1010} width={4} onValueChange={onChange} />)
    expect(screen.getAllByTestId(/^bit-toggle-/)).toHaveLength(4)
  })

  it('toggling a bit calls onValueChange with flipped value', () => {
    const onChange = vi.fn()
    render(<MultiBitInput nodeId="n1" currentValue={0b1010} width={4} onValueChange={onChange} />)
    fireEvent.click(screen.getByTestId('bit-toggle-0'))
    expect(onChange).toHaveBeenCalledWith('n1', 0b0010)
  })
})

describe('MultiBitInput (width > 8)', () => {
  it('renders numeric display for width=16', () => {
    render(<MultiBitInput nodeId="n1" currentValue={0x1234} width={16} onValueChange={vi.fn()} />)
    expect(screen.getByTestId('multibit-display')).toBeTruthy()
  })
})
```

## Pitfalls

- **Format state per node**: Each node independently tracks its display format. Use local component state (not Zustand) — format preference is a UI concern, not circuit state.
- **`parseValue` edge cases**: `parseInt('', 10)` → `NaN`. Handle gracefully — return `null`.
- **Bit toggle visual order**: MSB on the left (bit N-1), LSB on the right (bit 0). This matches conventional binary display.
- **`useState` is fine**: React Compiler allows `useState`. Don't replace with Zustand for local UI state.
- **PinoutPanel dependency**: If P05-10 hasn't merged yet, implement `MultiBitInput` standalone with full tests. Integration with PinoutPanel can happen later.
