## Context

The web-ide shows a `FullPinout` component with toggleable input pins and read-only output pins. HACER shows only circuit info (counts of gates, wires, inputs, outputs) in the sidebar — no pin-level detail.

Users need to:
- See all chip I/O pins with names, widths, and current values
- Toggle single-bit inputs (multi-bit editing is deferred)
- Click an "Eval" button to evaluate the circuit once without continuous simulation
- See output values update in real-time

### Current sidebar (`src/components/ui/Sidebar.tsx`):
Uses Ant Design `Layout.Sider` with `GateSelector`, `NodeSelector`, circuit info counts, simulation toggle, and controls. The PinoutPanel will be added as a new section in the sidebar.

### Current node data:
- `InputNode`: `{ id, name, position, rotation, value: number, width: number }` in `src/store/types.ts`
- `OutputNode`: same shape
- `updateInputNodeValue(nodeId, value: number)` in `nodeActions.ts`
- `simulationTick()` in `simulationActions.ts`

**Note:** The current store already uses numeric node values (`0`/`1` for single-bit, bitmask for bus widths).

### `src/components/ui/PinoutPanel.tsx`

```tsx
import { useCircuitStore } from '@/store/circuitStore'
import { circuitActions } from '@/store/circuitStore'
import { Button, Divider } from 'antd'

export function PinoutPanel() {
  const inputNodes = useCircuitStore(state => state.inputNodes)
  const outputNodes = useCircuitStore(state => state.outputNodes)

  const handleToggle = (nodeId: string, currentValue: number) => {
    const newValue = currentValue ? 0 : 1
    circuitActions.updateInputNodeValue(nodeId, newValue)
  }

  const handleEval = () => {
    circuitActions.simulationTick()
  }

  if (inputNodes.length === 0 && outputNodes.length === 0) {
    return null
  }

  return (
    <div data-testid="pinout-panel" style={{ padding: '8px 0' }}>
      <Divider style={{ margin: '8px 0', fontSize: 12 }}>Chip I/O</Divider>

      {inputNodes.length > 0 && (
        <div data-testid="pinout-inputs">
          <div style={{ fontWeight: 600, fontSize: 11, marginBottom: 4, color: '#888' }}>INPUTS</div>
          {inputNodes.map(node => (
            <div
              key={node.id}
              data-testid={`pin-input-${node.name}`}
              style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', padding: '2px 0' }}
            >
              <span style={{ fontSize: 12 }}>{node.name}{node.width > 1 ? `[${node.width}]` : ''}</span>
              <span
                style={{ cursor: node.width === 1 ? 'pointer' : 'default', fontFamily: 'monospace', fontSize: 12 }}
                onClick={() => node.width === 1 && handleToggle(node.id, node.value)}
                data-testid={`pin-toggle-${node.name}`}
              >
                {String(Number(node.value))}
              </span>
            </div>
          ))}
        </div>
      )}

      {outputNodes.length > 0 && (
        <div data-testid="pinout-outputs">
          <div style={{ fontWeight: 600, fontSize: 11, marginBottom: 4, marginTop: 8, color: '#888' }}>OUTPUTS</div>
          {outputNodes.map(node => (
            <div
              key={node.id}
              data-testid={`pin-output-${node.name}`}
              style={{ display: 'flex', justifyContent: 'space-between', padding: '2px 0' }}
            >
              <span style={{ fontSize: 12 }}>{node.name}{node.width > 1 ? `[${node.width}]` : ''}</span>
              <span style={{ fontFamily: 'monospace', fontSize: 12 }}>
                {String(Number(node.value))}
              </span>
            </div>
          ))}
        </div>
      )}

      <Button
        size="small"
        onClick={handleEval}
        style={{ marginTop: 8, width: '100%' }}
        data-testid="eval-button"
      >
        Eval
      </Button>
    </div>
  )
}
```

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest'
import { render, screen, fireEvent } from '@testing-library/react'

    simulationRunning: false,
    simulationSpeed: 100,
    lastSimulationError: null,

    junctionPreviewPosition: null,
    junctionPreviewWireId: null,
    statusMessages: [],
  })
}

beforeEach(() => {
  resetStore()
})

describe('PinoutPanel', () => {
  it('renders nothing when no nodes exist', () => {
    const { container } = render(<PinoutPanel />)
    expect(container.firstChild).toBeNull()
  })

  it('lists input nodes with names and values', () => {
    const s = useCircuitStore.getState()
    s.addInputNode('a', { x: 0, y: 0, z: 0 })
    s.addInputNode('b', { x: 0, y: 1, z: 0 })
    render(<PinoutPanel />)
    expect(screen.getByTestId('pin-input-a')).toBeTruthy()
    expect(screen.getByTestId('pin-input-b')).toBeTruthy()
  })

  it('lists output nodes with names and values', () => {
    const s = useCircuitStore.getState()
    s.addOutputNode('out', { x: 0, y: 0, z: 0 })
    render(<PinoutPanel />)
    expect(screen.getByTestId('pin-output-out')).toBeTruthy()
  })

  it('toggling input updates store value', () => {
    const s = useCircuitStore.getState()
    const node = s.addInputNode('a', { x: 0, y: 0, z: 0 })
    const before = node.value
    render(<PinoutPanel />)
    fireEvent.click(screen.getByTestId('pin-toggle-a'))
    const updated = useCircuitStore.getState().inputNodes.find(n => n.id === node.id)
    expect(updated?.value).toBe(before ? 0 : 1)
  })

  it('Eval button triggers simulationTick', () => {
    useCircuitStore.getState().addInputNode('a', { x: 0, y: 0, z: 0 })
    const spy = vi.spyOn(circuitActions, 'simulationTick')

    render(<PinoutPanel />)
    fireEvent.click(screen.getByTestId('eval-button'))

    expect(spy).toHaveBeenCalledTimes(1)
    spy.mockRestore()
  })

  it('shows bus width annotation for multi-bit nodes', () => {
    const s = useCircuitStore.getState()
    s.addInputNode('in', { x: 0, y: 0, z: 0 }, 16)
    render(<PinoutPanel />)
    expect(screen.getByTestId('pin-input-in').textContent).toContain('[16]')
  })
})
```

### `e2e/specs/simulation/pinout-panel.store.spec.ts`

```typescript
import { storeTest as test, storeExpect as expect } from '../../fixtures'

test.describe('Pinout Panel @store @simulation', () => {
  test('shows panel and toggles input value through the UI', async ({ page }) => {
    const inputNode = await page.evaluate(() => {
      const actions = window.__CIRCUIT_ACTIONS__
      if (!actions) throw new Error('Circuit actions are unavailable')
      actions.addOutputNode('out', { x: 4, y: 0, z: 0 })
      return actions.addInputNode('a', { x: 0, y: 0, z: 0 })
    })

    await expect(page.getByTestId('pinout-panel')).toBeVisible()
    await expect(page.getByTestId('pin-input-a')).toBeVisible()

    await page.getByTestId('pin-toggle-a').click()

    const nextValue = await page.evaluate((id: string) => {
      const node = window.__CIRCUIT_STORE__?.inputNodes.find((n) => n.id === id)
      return node?.value
    }, inputNode.id)

    expect(nextValue).toBe(0)
  })
})
```

## Pitfalls

- **Antd mock in tests**: Prefer mocking only what you need (`Button`, `Divider`) and **`import { vi } from 'vitest'`**. Alternatively render with real Ant Design inside `ConfigProvider` if jsdom styling is acceptable — stay consistent with nearby UI tests.
- **Value type**: Node values are numeric in current store (`number`). Use `String(Number(node.value))` for compact display.
- **`updateInputNodeValue` param type**: Expects `number`. Toggle logic should pass numeric `0`/`1`.
- **Sidebar width**: The sidebar is an Ant Design `Sider` with limited width. Keep the panel compact — short labels, monospace values, no unnecessary padding.
- **React Compiler**: No `useMemo`, `useCallback`, `React.memo`. Just use `useCircuitStore` selectors.
- **Multi-bit editing**: For this ticket, multi-bit inputs just show the numeric value. Full bit-toggle array and format selector come in P05-13 (Layer 1). For now, a simple display is sufficient.
- **`simulationTick` for Eval**: Calling `simulationTick()` once is the "eval" behavior. After P05-03 (topological sort), this gives correct results in one call. If P05-03 hasn't run yet, multi-layer circuits may need multiple clicks — that's acceptable and will be fixed by P05-03.
