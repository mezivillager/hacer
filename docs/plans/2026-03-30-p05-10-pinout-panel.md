# P05-10 PinoutPanel Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a sidebar PinoutPanel that shows chip I/O values, allows single-bit input toggling, and provides one-click Eval via `simulationTick`.

**Architecture:** Implement a focused UI component (`PinoutPanel`) that reads `inputNodes`/`outputNodes` from Zustand selectors and dispatches actions through `circuitActions`. Keep the panel compact for the existing Ant Design `Sider` layout, and gate toggling to single-bit nodes (`width === 1`) while displaying all values as numbers. Validate behavior with component tests plus a Playwright store spec that verifies UI-triggered state updates.

**Tech Stack:** React 19, TypeScript strict, Zustand + Immer, Ant Design, Vitest, Playwright store tests.

```tsx
import { useCircuitStore, circuitActions } from '@/store/circuitStore'
import { Button, Divider } from 'antd'

export function PinoutPanel() {
  const inputNodes = useCircuitStore((state) => state.inputNodes)
  const outputNodes = useCircuitStore((state) => state.outputNodes)

  const handleToggle = (nodeId: string, currentValue: number) => {
    const newValue = currentValue ? 0 : 1
    circuitActions.updateInputNodeValue(nodeId, newValue)
  }

  const handleEval = () => {
    circuitActions.simulationTick()
  }

  if (inputNodes.length === 0 && outputNodes.length === 0) {
    return null
  }

  return (
    <div data-testid="pinout-panel" style={{ padding: '8px 0' }}>
      <Divider style={{ margin: '8px 0', fontSize: 12 }}>Chip I/O</Divider>

      {inputNodes.length > 0 && (
        <div data-testid="pinout-inputs">
          <div style={{ fontWeight: 600, fontSize: 11, marginBottom: 4, color: '#888' }}>INPUTS</div>
          {inputNodes.map((node) => (
            <div
              key={node.id}
              data-testid={`pin-input-${node.name}`}
              style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', padding: '2px 0' }}
            >
              <span style={{ fontSize: 12 }}>
                {node.name}
                {node.width > 1 ? `[${node.width}]` : ''}
              </span>
              <span
                style={{ cursor: node.width === 1 ? 'pointer' : 'default', fontFamily: 'monospace', fontSize: 12 }}
                onClick={() => node.width === 1 && handleToggle(node.id, node.value)}
                data-testid={`pin-toggle-${node.name}`}
              >
                {String(Number(node.value))}
              </span>
            </div>
          ))}
        </div>
      )}

      {outputNodes.length > 0 && (
        <div data-testid="pinout-outputs">
          <div style={{ fontWeight: 600, fontSize: 11, marginBottom: 4, marginTop: 8, color: '#888' }}>OUTPUTS</div>
          {outputNodes.map((node) => (
            <div
              key={node.id}
              data-testid={`pin-output-${node.name}`}
              style={{ display: 'flex', justifyContent: 'space-between', padding: '2px 0' }}
            >
              <span style={{ fontSize: 12 }}>
                {node.name}
                {node.width > 1 ? `[${node.width}]` : ''}
              </span>
              <span style={{ fontFamily: 'monospace', fontSize: 12 }}>{String(Number(node.value))}</span>
            </div>
          ))}
        </div>
      )}

      <Button size="small" onClick={handleEval} style={{ marginTop: 8, width: '100%' }} data-testid="eval-button">
        Eval
      </Button>
    </div>
  )
}
```

import { storeTest as test, storeExpect as expect } from '../../fixtures'

test.describe('Pinout Panel @store @simulation', () => {
  test('shows panel and updates input node value from UI toggle', async ({ page }) => {
    const setup = await page.evaluate(() => {
      const actions = window.__CIRCUIT_ACTIONS__
      if (!actions) {
        throw new Error('Circuit actions are unavailable')
      }

      actions.addOutputNode('out', { x: 4, y: 0, z: 0 })
      const input = actions.addInputNode('a', { x: 0, y: 0, z: 0 })
      return { id: input.id, before: input.value }
    })

    await expect(page.getByTestId('pinout-panel')).toBeVisible()
    await expect(page.getByTestId('pin-input-a')).toBeVisible()

    await page.getByTestId('pin-toggle-a').click()

    const after = await page.evaluate((id: string) => {
      return window.__CIRCUIT_STORE__?.inputNodes.find((n) => n.id === id)?.value
    }, setup.id)

    expect(after).toBe(setup.before ? 0 : 1)
  })
})
