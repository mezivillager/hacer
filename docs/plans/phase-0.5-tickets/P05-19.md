## Context

HACER should guide users through chip construction workflows without hard-coupling runtime architecture to curriculum bundle APIs. Users start from a blank canvas today and need a capability-first browser to select target chips, inspect interfaces, and bootstrap I/O.

P05-01 provides the `ChipRegistry` for chip interface lookups. Test/template assets can come from fixtures or pluggable providers.

### Current sidebar (`src/components/ui/Sidebar.tsx`):

The sidebar has sections: Elementary Gates (GateSelector), Circuit I/O (NodeSelector), Controls, Circuit Info. The chip workflow browser will be a new section.

### Chip interface reference (from gap-analysis.md):

| Chip | Inputs | Outputs |
|------|--------|---------|
| Not | `in` | `out` |
| And | `a, b` | `out` |
| Mux | `a, b, sel` | `out` |
| Not16 | `in[16]` | `out[16]` |
| Mux4Way16 | `a[16], b[16], c[16], d[16], sel[2]` | `out[16]` |
| DMux8Way | `in, sel[3]` | `a, b, c, d, e, f, g, h` |

## Files to Create

### `src/components/ui/ChipWorkflowBrowser.tsx`

```tsx
import { circuitActions } from '@/store/circuitStore'
import { Button, Divider, List, Typography } from 'antd'
import { PROJECT1_CHIP_SEQUENCE } from '@/core/testing/project1ChipOrder'
import { useState } from 'react'

const { Text } = Typography

// Prefer one module, e.g. `src/core/testing/project1ChipSpecs.ts`, exporting `CHIP_SPECS`
// for all **15** `PROJECT1_CHIP_SEQUENCE` names (Not … DMux8Way) — do not leave `// ...` at ship time.
const CHIP_SPECS: Record<string, { inputs: Array<{ name: string; width: number }>; outputs: Array<{ name: string; width: number }>; description: string }> = {
  Not: { inputs: [{ name: 'in', width: 1 }], outputs: [{ name: 'out', width: 1 }], description: 'Negation: out = NOT in' },
  And: { inputs: [{ name: 'a', width: 1 }, { name: 'b', width: 1 }], outputs: [{ name: 'out', width: 1 }], description: 'And: out = a AND b' },
  // … remaining 13 chips (see gap-analysis pin tables)
}

export function ChipWorkflowBrowser() {
  const [selectedChip, setSelectedChip] = useState<string | null>(null)
  const [completedChips, setCompletedChips] = useState<Set<string>>(() => {
    try {
      const saved = localStorage.getItem('hacer-completed-chips')
      return saved ? new Set(JSON.parse(saved)) : new Set()
    } catch { return new Set() }
  })

  const handleStart = (chipName: string) => {
    const spec = CHIP_SPECS[chipName]
    if (!spec) return

    circuitActions.clearCircuit()
    // P05-20: also `circuitActions.setCurrentChip(...)` with a `ChipDefinition` stub (interface-only) so ChipDefinitionPanel / validation stay in sync.

    // Create input nodes on the left side
    spec.inputs.forEach((pin, i) => {
      circuitActions.addInputNode(pin.name, { x: -4, y: 0, z: i * 4 }, pin.width)
    })

    // Create output nodes on the right side
    spec.outputs.forEach((pin, i) => {
      circuitActions.addOutputNode(pin.name, { x: 8, y: 0, z: i * 4 }, pin.width)
    })

    setSelectedChip(chipName)
  }

  return (
    <div data-testid="chip-workflow-browser">
      <Divider style={{ margin: '8px 0', fontSize: 12 }}>Project 1: Boolean Logic</Divider>
      <List
        size="small"
        dataSource={PROJECT1_CHIP_SEQUENCE}
        renderItem={(chipName) => {
          const isCompleted = completedChips.has(chipName)
          const isSelected = selectedChip === chipName
          return (
            <List.Item
              data-testid={`chip-item-${chipName}`}
              style={{ cursor: 'pointer', padding: '4px 8px', background: isSelected ? '#1a1a2e' : 'transparent' }}
              onClick={() => setSelectedChip(chipName)}
            >
              <Text style={{ fontSize: 12 }}>
                {isCompleted ? '✓ ' : '○ '}{chipName}
              </Text>
            </List.Item>
          )
        }}
      />
      {selectedChip && CHIP_SPECS[selectedChip] && (
        <div data-testid="chip-spec" style={{ padding: '8px', fontSize: 11 }}>
          <Text strong style={{ fontSize: 12 }}>{selectedChip}</Text>
          <div style={{ color: '#888', marginTop: 4 }}>{CHIP_SPECS[selectedChip].description}</div>
          <div style={{ marginTop: 4 }}>
            <Text type="secondary">IN: </Text>
            {CHIP_SPECS[selectedChip].inputs.map(p =>
              `${p.name}${p.width > 1 ? `[${p.width}]` : ''}`
            ).join(', ')}
          </div>
          <div>
            <Text type="secondary">OUT: </Text>
            {CHIP_SPECS[selectedChip].outputs.map(p =>
              `${p.name}${p.width > 1 ? `[${p.width}]` : ''}`
            ).join(', ')}
          </div>
          <Button
            size="small"
            type="primary"
            block
            style={{ marginTop: 8 }}
            onClick={() => handleStart(selectedChip)}
            data-testid="chip-start-button"
          >
            Start {selectedChip}
          </Button>
        </div>
      )}
    </div>
  )
}
```

### `src/components/ui/ChipWorkflowBrowser.test.tsx`

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest'
import type { ReactNode } from 'react'
import { render, screen, fireEvent } from '@testing-library/react'
import { useCircuitStore } from '@/store/circuitStore'
import { ChipWorkflowBrowser } from './ChipWorkflowBrowser'

function MockList({ dataSource, renderItem }: { dataSource: string[]; renderItem: (item: string) => ReactNode }) {
  return <ul>{dataSource.map((item, i) => <li key={i}>{renderItem(item)}</li>)}</ul>
}
MockList.Item = ({ children, ...props }: { children?: ReactNode }) => <div {...props}>{children}</div>

vi.mock('antd', () => ({
  Button: ({ children, onClick, ...props }: any) => (
    <button type="button" onClick={onClick} {...props}>{children}</button>
  ),
  Divider: ({ children }: any) => <hr />,
  List: MockList,
  Typography: { Text: ({ children, ...props }: any) => <span {...props}>{children}</span> },
}))

function resetStore() {
  // Zustand: no `getInitialState()` — mirror `initialState` in `src/store/circuitStore.ts`
  // (add `currentChipDefinition: null` here when P05-20 lands; add other slices as tickets merge).
  useCircuitStore.setState({
    gates: [], wires: [], selectedGateId: null, selectedWireId: null,
    simulationRunning: false, simulationSpeed: 100, placementMode: null,
    placementPreviewPosition: null, wiringFrom: null, isDragActive: false,
    hoveredGateId: null, showAxes: false, inputNodes: [], outputNodes: [],
    junctions: [], nodePlacementMode: null, selectedNodeId: null,
    selectedNodeType: null, junctionPlacementMode: null,
    junctionPreviewPosition: null, junctionPreviewWireId: null,
  })
}

beforeEach(() => {
  resetStore()
  localStorage.clear()
})

describe('ChipWorkflowBrowser', () => {
  it('renders chip list with 15 items', () => {
    render(<ChipWorkflowBrowser />)
    // PROJECT1_CHIP_SEQUENCE has 15 entries (Not through DMux8Way)
    const items = screen.getAllByTestId(/^chip-item-/)
    expect(items.length).toBe(15)
  })

  it('clicking a chip shows its spec', () => {
    render(<ChipWorkflowBrowser />)
    fireEvent.click(screen.getByTestId('chip-item-Not'))
    expect(screen.getByTestId('chip-spec')).toBeTruthy()
    expect(screen.getByTestId('chip-spec').textContent).toContain('in')
    expect(screen.getByTestId('chip-spec').textContent).toContain('out')
  })

  it('clicking Start creates correct I/O nodes', () => {
    render(<ChipWorkflowBrowser />)
    fireEvent.click(screen.getByTestId('chip-item-Not'))
    fireEvent.click(screen.getByTestId('chip-start-button'))

    const state = useCircuitStore.getState()
    expect(state.inputNodes).toHaveLength(1)
    expect(state.inputNodes[0].name).toBe('in')
    expect(state.outputNodes).toHaveLength(1)
    expect(state.outputNodes[0].name).toBe('out')
  })

  it('clicking Start for Mux creates 3 inputs + 1 output', () => {
    render(<ChipWorkflowBrowser />)
    fireEvent.click(screen.getByTestId('chip-item-Mux'))
    fireEvent.click(screen.getByTestId('chip-start-button'))

    const state = useCircuitStore.getState()
    expect(state.inputNodes).toHaveLength(3)
    expect(state.inputNodes.map(n => n.name).sort()).toEqual(['a', 'b', 'sel'])
    expect(state.outputNodes).toHaveLength(1)
    expect(state.outputNodes[0].name).toBe('out')
  })
})
```

## Pitfalls

- **`CHIP_SPECS` data source**: Hardcoding pin interfaces is simplest and avoids runtime parsing. All 15 interfaces are in the gap-analysis doc. If a trusted fixture/provider exists, you can generate this map at build time.
- **`clearCircuit` before Start**: Warn the user before clearing an unsaved circuit. For Layer 2, a simple confirmation is sufficient.
- **Node positioning**: Position inputs on the left, outputs on the right, spaced vertically by `4` grid units. For chips with many pins (DMux8Way has 8 outputs), ensure spacing doesn't overflow the visible canvas.
- **Antd mock in tests**: Mock `List`, `List.Item`, `Button`, `Divider`, `Typography` — follow the established pattern from P05-10.
- **Completion tracking**: Store as `Set<string>` serialized to `localStorage`. Update when a test passes (listener in P05-22).
- **React Compiler**: `useState` for selectedChip and completedChips is fine. No `useMemo`.
