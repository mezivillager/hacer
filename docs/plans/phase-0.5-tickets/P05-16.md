## Context

P05-04 created the HDL parser: `parseHDL(source) → HDLChip` AST. P05-01 created the `ChipRegistry`. This ticket bridges them: compile an `HDLChip` AST into a `ChipDefinition` with `implementation.type === 'hdl'` that can be evaluated.

### HDL compilation pipeline:

```
HDL source → parseHDL() → HDLChip AST → compileHDL() → ChipDefinition → evaluate → results
```

### Key concepts:

1. **Part resolution**: Each `PARTS:` entry names a chip (e.g., `Nand(a=a, b=b, out=nandOut)`). The compiler looks up `'Nand'` in the registry.
2. **Internal signals**: Connections like `out=nandOut` create internal wires. `nandOut` is neither a chip input nor output — it's a signal internal to the implementation.
3. **Pin mapping**: `a=a` maps the part's pin `a` to the chip's input `a`. `out=out` maps the part's output to the chip's output.
4. **Topological ordering**: Parts must be evaluated in dependency order (same principle as P05-03 for gates).
5. **Sub-bus**: `a[0]` or `a[0..7]` accesses individual bits — needed for 16-bit chips.

### Not.hdl example:

```hdl
CHIP Not {
    IN in;
    OUT out;
    PARTS:
    Nand(a=in, b=in, out=out);
}
```

### And.hdl example (2-part, uses internal wire):

```hdl
CHIP And {
    IN a, b;
    OUT out;
    PARTS:
    Nand(a=a, b=b, out=nandOut);
    Not(in=nandOut, out=out);
}
```

## Files to Create

### `src/core/hdl/compiler.ts`

```typescript
import type { HDLChip, HDLPart, HDLConnection } from './types'
import type { ChipDefinition, ChipPin, BuiltinEvalFn } from '../chips/types'
import type { ChipRegistry } from '../chips/registry'
import { isBuiltinChip } from '../chips/types'

export interface HDLCompileError {
  message: string
  partName?: string
  pinName?: string
}

export type HDLCompileResult =
  | { type: 'success'; chip: ChipDefinition }
  | { type: 'error'; errors: HDLCompileError[] }

export function compileHDL(ast: HDLChip, registry: ChipRegistry): HDLCompileResult {
  const errors: HDLCompileError[] = []

  // 1. Handle BUILTIN declaration
  if (ast.builtin) {
    // The chip IS a builtin — it should already be in the registry
    const existing = registry.get(ast.name)
    if (existing) return { type: 'success', chip: existing }
    errors.push({ message: `Builtin chip "${ast.name}" not found in registry` })
    return { type: 'error', errors }
  }

  // 2. Resolve all parts
  const resolvedParts: Array<{ part: HDLPart; chipDef: ChipDefinition }> = []
  for (const part of ast.parts) {
    const partChip = registry.get(part.name)
    if (!partChip) {
      errors.push({ message: `Unknown chip-part: "${part.name}"`, partName: part.name })
      continue
    }
    resolvedParts.push({ part, chipDef: partChip })
  }
  if (errors.length > 0) return { type: 'error', errors }

  // 3. Build internal signal graph
  // Map: signalName → { producers: [...], consumers: [...] }
  // Chip inputs are sources; chip outputs are sinks; everything else is internal

  // 4. Validate: all part pins connected, no dangling signals, width compatibility

  // 5. Determine topological order of parts

  // 6. Build evaluate function
  const chipInputs: ChipPin[] = ast.inputs.map(p => ({ name: p.name, width: p.width ?? 1 }))
  const chipOutputs: ChipPin[] = ast.outputs.map(p => ({ name: p.name, width: p.width ?? 1 }))

  const evaluate: BuiltinEvalFn = (inputs) => {
    // Initialize signal map with chip inputs
    const signals: Record<string, number> = { ...inputs }

    // Evaluate parts in topological order
    for (const { part, chipDef } of resolvedParts) {
      // Build part inputs from signal map using connection mappings
      const partInputs: Record<string, number> = {}
      for (const conn of part.connections) {
        // conn.internal = part pin name, conn.external = signal name (chip I/O or internal)
        const partPin = chipDef.inputs.find(p => p.name === conn.internal)
        if (partPin) {
          partInputs[conn.internal] = signals[conn.external] ?? 0
        }
      }

      // Evaluate the part
      let partOutputs: Record<string, number>
      if (isBuiltinChip(chipDef)) {
        partOutputs = chipDef.implementation.evaluate(partInputs)
      } else {
        // Recursive: evaluate HDL chip (via evaluateHDLChip or similar)
        partOutputs = evaluate.call(null, partInputs) // placeholder — needs proper recursion
      }

      // Write part outputs to signal map
      for (const conn of part.connections) {
        const partOutputPin = chipDef.outputs.find(p => p.name === conn.internal)
        if (partOutputPin && partOutputs[conn.internal] !== undefined) {
          signals[conn.external] = partOutputs[conn.internal]
        }
      }
    }

    // Collect chip outputs from signal map
    const result: Record<string, number> = {}
    for (const out of chipOutputs) {
      result[out.name] = signals[out.name] ?? 0
    }
    return result
  }

  const chipDef: ChipDefinition = {
    name: ast.name,
    inputs: chipInputs,
    outputs: chipOutputs,
    implementation: { type: 'builtin', evaluate },
    // Note: compiled HDL chips use 'builtin' type with a generated evaluate function
    // rather than 'hdl' type, because they are directly evaluable
  }

  return { type: 'success', chip: chipDef }
}
```

**Design note:** The compiled chip stores a closure that captures the resolved parts and evaluates them in order. This is stored as `type: 'builtin'` with a generated evaluate function (simpler than maintaining a separate evaluation path for HDL chips). Alternative: use `type: 'hdl'` and implement a separate `evaluateHDLChip` — discuss trade-offs at implementation time.

### `src/core/hdl/compiler.test.ts`

```typescript
import { describe, it, expect, beforeEach } from 'vitest'
import { createChipRegistry, registerBuiltin } from '../chips/registry'
import type { ChipRegistry } from '../chips/registry'
import { parseHDL } from './parser'
import { compileHDL } from './compiler'
import { isBuiltinChip } from '../chips/types'

let registry: ChipRegistry

beforeEach(() => {
  registry = createChipRegistry()
  // Register Nand as the primitive
  registerBuiltin(registry, 'Nand',
    [{ name: 'a', width: 1 }, { name: 'b', width: 1 }],
    [{ name: 'out', width: 1 }],
    (i) => ({ out: (i.a & i.b) === 0 ? 1 : 0 })
  )
})

describe('compileHDL', () => {
  it('compiles Not.hdl and evaluates correctly', () => {
    const ast = parseHDL(`
      CHIP Not { IN in; OUT out;
        PARTS: Nand(a=in, b=in, out=out);
      }
    `)
    expect(ast.type).toBe('success')
    if (ast.type !== 'success') return

    const result = compileHDL(ast.chip, registry)
    expect(result.type).toBe('success')
    if (result.type !== 'success') return

    const chip = result.chip
    expect(isBuiltinChip(chip)).toBe(true)
    if (!isBuiltinChip(chip)) return

    expect(chip.implementation.evaluate({ in: 0 })).toEqual({ out: 1 })
    expect(chip.implementation.evaluate({ in: 1 })).toEqual({ out: 0 })
  })

  it('compiles And.hdl with internal wire', () => {
    // Register Not first (compiled from HDL or as builtin)
    registerBuiltin(registry, 'Not',
      [{ name: 'in', width: 1 }],
      [{ name: 'out', width: 1 }],
      (i) => ({ out: i.in === 0 ? 1 : 0 })
    )

    const ast = parseHDL(`
      CHIP And { IN a, b; OUT out;
        PARTS:
        Nand(a=a, b=b, out=nandOut);
        Not(in=nandOut, out=out);
      }
    `)
    expect(ast.type).toBe('success')
    if (ast.type !== 'success') return

    const result = compileHDL(ast.chip, registry)
    expect(result.type).toBe('success')
    if (result.type !== 'success') return

    const chip = result.chip
    if (!isBuiltinChip(chip)) return

    expect(chip.implementation.evaluate({ a: 0, b: 0 })).toEqual({ out: 0 })
    expect(chip.implementation.evaluate({ a: 0, b: 1 })).toEqual({ out: 0 })
    expect(chip.implementation.evaluate({ a: 1, b: 0 })).toEqual({ out: 0 })
    expect(chip.implementation.evaluate({ a: 1, b: 1 })).toEqual({ out: 1 })
  })

  it('reports error for unknown chip-part', () => {
    const ast = parseHDL(`
      CHIP Foo { IN a; OUT out;
        PARTS: UnknownChip(a=a, out=out);
      }
    `)
    if (ast.type !== 'success') return

    const result = compileHDL(ast.chip, registry)
    expect(result.type).toBe('error')
    if (result.type === 'error') {
      expect(result.errors[0].message).toContain('UnknownChip')
    }
  })

  it('handles BUILTIN declaration', () => {
    const ast = parseHDL('CHIP Nand { IN a, b; OUT out; BUILTIN Nand; }')
    if (ast.type !== 'success') return

    const result = compileHDL(ast.chip, registry)
    expect(result.type).toBe('success')
  })
})
```

## Pitfalls

- **Part evaluation order**: Parts must be topologically sorted by signal dependencies. If `Not(in=nandOut, ...)` depends on `Nand(..., out=nandOut)`, Nand must evaluate first. Use a simple Kahn's algorithm over the part dependency graph.
- **HDLChip AST shape**: Depends on P05-04's exact AST types. The test file uses `parseHDL` — if the parser returns `{ type: 'success', chip: HDLChip }`, match that shape. Read P05-04 AST types before implementing.
- **Connection direction**: In `Nand(a=in, b=in, out=nandOut)`, `a` and `b` are part **inputs** (read from signal `in`), `out` is a part **output** (writes to signal `nandOut`). The compiler must distinguish by checking the part's `ChipDefinition.inputs` vs `.outputs`.
- **Sub-bus connections**: `a[0]` in `Not(in=in[0], out=out[0])` means "connect to bit 0 of signal `in`". Use `readSubBus`/`writeSubBus` from P05-11 (or inline bit ops). If P05-11 hasn't merged, implement the bit extraction inline.
- **`true`/`false` literals**: HDL allows `Nand(a=true, b=in, ...)`. The parser produces these as special connection values. The compiler must inject constant 1/0 instead of reading from the signal map.
- **Recursive evaluation**: When a compiled HDL chip is used as a part in another HDL chip, the evaluate closure must call itself recursively. This works naturally if compiled chips are stored as `type: 'builtin'` with closures. Max depth guard recommended (100 levels).
